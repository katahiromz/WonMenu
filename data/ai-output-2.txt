+/**********************************************************************
+ *         IntGetMenuState / IntFreeMenuState
+ *
+ * pti->pMenuState を必要なら確保する/解放するだけの最小限のヘルパー。
+ * Windows 本来の実装(xxxMNAllocMenuState 等)はロックカウントやアニメ
+ * ーション用 DC、入れ子のメニューループ用スタックまで管理しているが、
+ * ここでは MENU_InitTracking〜MENU_ExitTracking の1回のループ分だけ
+ * pGlobalPopupMenu を保持できれば十分なので、その最小実装にとどめる。
+ */
+static PMENUSTATE FASTCALL IntGetMenuState(PTHREADINFO pti, struct _DESKTOP *rpdesk, BOOL bCreate)
+{
+   PMENUSTATE pMenuState = pti->pMenuState;
+
+   if (pMenuState || !bCreate)
+      return pMenuState;
+
+   pMenuState = DesktopHeapAlloc(rpdesk, sizeof(MENUSTATE));
+   if (!pMenuState)
+      return NULL;
+
+   RtlZeroMemory(pMenuState, sizeof(MENUSTATE));
+   pMenuState->ptiMenuStateOwner = pti;
+   pti->pMenuState = pMenuState;
+
+   return pMenuState;
+}
+
+static VOID FASTCALL IntFreeMenuState(PTHREADINFO pti, struct _DESKTOP *rpdesk)
+{
+   PMENUSTATE pMenuState = pti->pMenuState;
+
+   if (!pMenuState)
+      return;
+
+   /* 入れ子ループは未対応なので常に pmnsPrev (NULL) に戻すだけ。 */
+   pti->pMenuState = pMenuState->pmnsPrev;
+   DesktopHeapFree(rpdesk, pMenuState);
+}
