+ *         IntGetMSWND

+ * 返す。プロセス全体で1つしかない static HWND (top_popup) ではなく、

+ * メニューループが同時に走るケースでも正しく動作する。
+ */
+static PWND FASTCALL IntGetMSWND(PMENUSTATE pMenuState)
+{
+   PPOPUPMENU pPopupMenu;
+
+   if (!pMenuState || !pMenuState->pGlobalPopupMenu)
+      return NULL;
+
+   pPopupMenu = pMenuState->pGlobalPopupMenu;
+
+   if (pMenuState->fInsideMenuLoop)
+      return pPopupMenu->spwndPopupMenu;
+
+   if (pPopupMenu->spwndNextPopup)
+      return pPopupMenu->spwndNextPopup;
+
+   return pPopupMenu->spwndActivePopup;
+}
